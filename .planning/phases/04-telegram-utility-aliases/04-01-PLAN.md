// ── Telegram HTML formatter ─────────────────────────────────────────────────
function formatHtml(text: string): string {
  // Truncate to Telegram's 4096 char limit
  if (text.length > 4000) return text.slice(0, 4000) + "\n\n<i>…truncated</i>";
  return text;
}

function toolsPath(): string {
  return process.env.GSD_TOOLS_PATH
    ?? join(import.meta.dirname, "..", "bin", "gsd-tools.cjs");
}

function noProject(cmd: string): string {
  return formatHtml(
    `⚠️ No GSD project found.\n\nCurrent directory: <code>${process.cwd()}</code>\n\nTo start a new project here:\n<b>/${cmd.replace(/_/g, '_')}</b> is for an initialized project.\nRun <b>/gsd_new_project</b> to initialize.\n\nOr navigate to an existing GSD project directory first.`
  );
}

api.registerCommand({
  name: "gsd_status",
  description: "Show GSD project status — current phase, progress, milestone",
  acceptsArgs: false,
  requireAuth: false,
  handler() {
    try {
      const snap = execGsdTools("state-snapshot") as Record<string, unknown>;
      if ((snap as any).error) return { text: noProject("gsd_new_project") };
      const state = (snap as any).state ?? snap;
      const cfg = (state as any).config ?? {};
      const milestone = cfg.milestone ?? "v?";
      const milestoneName = cfg.milestone_name ?? "";
      const progress = (state as any).progress ?? {};
      const total = progress.total_plans ?? 0;
      const done = progress.completed_plans ?? 0;
      const pct = total > 0 ? Math.round((done / total) * 100) : 0;
      const barLen = 10;
      const filled = Math.round((pct / 100) * barLen);
      const bar = "█".repeat(filled) + "░".repeat(barLen - filled);
      const phase = cfg.current_phase ?? progress.current_phase ?? "—";
      const lines = [
        `<b>GSD Project Status</b>`,
        ``,
        `Milestone: <b>${milestone}${milestoneName ? " " + milestoneName : ""}</b>`,
        `Phase: <code>${phase}</code>`,
        `Progress: <code>[${bar}] ${done}/${total} plans (${pct}%)</code>`,
      ];
      return { text: formatHtml(lines.join("\n")) };
    } catch (e) {
      return { text: `<b>Error:</b> <code>${String(e)}</code>` };
    }
  },
});

api.registerCommand({
  name: "gsd_health",
  description: "Run GSD project health check",
  acceptsArgs: false,
  requireAuth: false,
  handler() {
    try {
      const result = execGsdTools("validate health") as Record<string, unknown>;
      if ((result as any).error) return { text: noProject("gsd_new_project") };
      const healthy = (result as any).healthy ?? (result as any).valid ?? false;
      const issues = ((result as any).issues ?? []) as string[];
      const warnings = ((result as any).warnings ?? []) as string[];
      const lines = [
        `<b>GSD Health Check</b>`,
        ``,
        healthy ? `✅ Project is healthy` : `❌ Issues found`,
        ...issues.map(i => `• <code>${i}</code>`),
        ...(warnings.length > 0 ? [``, `⚠️ Warnings:`] : []),
        ...warnings.map(w => `• ${w}`),
      ];
      return { text: formatHtml(lines.join("\n")) };
    } catch (e) {
      return { text: noProject("gsd_new_project") };
    }
  },
});

api.registerCommand({
  name: "gsd_cleanup",
  description: "Clean up GSD artifacts (temp files, old research)",
  acceptsArgs: false,
  requireAuth: false,
  handler() {
    try {
      // Remove .planning/research/ temp files if they exist
      const researchDir = join(process.cwd(), ".planning", "research");
      let cleaned: string[] = [];
      try {
        const { readdirSync, unlinkSync, statSync } = require("fs") as typeof import("fs");
        const files = readdirSync(researchDir).filter((f: string) => f.endsWith(".md"));
        for (const f of files) {
          const fp = join(researchDir, f);
          if (statSync(fp).isFile()) { unlinkSync(fp); cleaned.push(f); }
        }
      } catch { /* research dir may not exist */ }
      const lines = [
        `<b>GSD Cleanup</b>`,
        ``,
        cleaned.length > 0
          ? `✅ Removed ${cleaned.length} temp file(s):\n${cleaned.map(f => `• <code>${f}</code>`).join("\n")}`
          : `✅ Nothing to clean up — project is tidy`,
      ];
      return { text: formatHtml(lines.join("\n")) };
    } catch (e) {
      return { text: `<b>Error:</b> <code>${String(e)}</code>` };
    }
  },
});
