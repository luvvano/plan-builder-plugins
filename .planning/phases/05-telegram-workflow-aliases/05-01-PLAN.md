**Script to execute:**

```bash
cd /home/egor/projects/plan-builder-plugins/openclaw-plugin

SKILLS_SRC="skills/workflows"
SKILLS_DST="skills/telegram-aliases"

declare -A MAP=(
  ["quick"]="gsd_quick"
  ["new-project"]="gsd_new_project"
  ["plan-phase"]="gsd_plan_phase"
  ["execute-phase"]="gsd_execute_phase"
  ["discuss-phase"]="gsd_discuss_phase"
  ["verify-work"]="gsd_verify_work"
  ["add-phase"]="gsd_add_phase"
  ["insert-phase"]="gsd_insert_phase"
  ["remove-phase"]="gsd_remove_phase"
  ["new-milestone"]="gsd_new_milestone"
  ["complete-milestone"]="gsd_complete_milestone"
  ["resume-work"]="gsd_resume_work"
  ["pause-work"]="gsd_pause_work"
  ["debug"]="gsd_debug"
  ["add-todo"]="gsd_add_todo"
  ["check-todos"]="gsd_check_todos"
  ["audit-milestone"]="gsd_audit_milestone"
  ["add-tests"]="gsd_add_tests"
)

for slug in "${!MAP[@]}"; do
  alias="${MAP[$slug]}"
  src="$SKILLS_SRC/$slug/SKILL.md"
  dst_dir="$SKILLS_DST/$slug"
  dst="$dst_dir/SKILL.md"
  
  mkdir -p "$dst_dir"
  
  # Copy source, replace only the name: field in frontmatter
  python3 -c "
import sys, re
src = open('$src').read()
# Replace name: gsd:* with name: gsd_*
new = re.sub(r'^name: gsd:[^\n]+', 'name: $alias', src, count=1, flags=re.MULTILINE)
open('$dst', 'w').write(new)
print('Created $dst')
"
done

echo "Done. Count:"
ls "$SKILLS_DST" | wc -l
```
